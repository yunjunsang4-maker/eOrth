## 사용법

```tsx
import { GlobeIcon, CameraIcon, TrashIcon } from '../components/icons';

// 기본 (보라 그라데이션)
<GlobeIcon size={24} />

// 단색 지정
<CameraIcon size={20} color="#A1A1B0" />

// 알림 닷 표시
<BellIcon size={24} dot dotColor="#FF4D4D" />
```
